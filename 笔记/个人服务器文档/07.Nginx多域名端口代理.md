创建域名指向IP的解析，只需要解析，无需代理
![[Pasted image 20251120190622.png]]

```
vi /etc/nginx/sites-available/music
```

```
server {
    listen 80;
    server_name music.mlishu.site;

    location / {
        proxy_pass http://127.0.0.1:4533;  # 将请求转发到本地 4533 端口
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

```

```
ln -s /etc/nginx/sites-available/music /etc/nginx/sites-enabled/
```

```
nginx -t
```

```
server {
    server_name music.mlishu.site;

    location / {
        proxy_pass http://127.0.0.1:4533;  # 将请求转发到本地 4533 端口
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/music.mlishu.site/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/music.mlishu.site/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}

server {
    if ($host = music.mlishu.site) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name music.mlishu.site;
    return 404; # managed by Certbot
}
```
